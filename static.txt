//1.Define a static variable and access that through a class


class MyClass:
    static_var = "Hello, I am a static variable"
print(MyClass.static_var)
obj1, obj2 = MyClass(), MyClass()
print(obj1.static_var)
MyClass.static_var = "New value"
print(obj1.static_var, obj2.static_var)


//2.Define a static variable and access that through a instance using python


class MyClass:
    static_var = "Hello, I am a static variable"
obj = MyClass()
print(obj.static_var) 



//3.Define a static variable and change within the instance using python



class MyClass:
    static_var = "Hello, I am a static variable"
obj1, obj2 = MyClass(), MyClass()
obj1.static_var = "Modified via instance"
print(obj1.static_var, obj2.static_var, MyClass.static_var)



//4.Define a static variable and change within the class


class MyClass:
    static_var = 10
print("Initial Value:", MyClass.static_var)
MyClass.static_var = 20
print("Modified Value:", MyClass.static_var)
